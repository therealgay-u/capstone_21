pipeline {
    agent any

    environment {
        PORT = '3000'
    }

    stages {

        stage('Checkout') {
            steps {
                checkout scm
            }
        }

        stage('Install Dependencies') {
            steps {
                bat 'npm install'
            }
        }

        stage('Start Application') {
            steps {
                // Start app in background
                bat '''
                start /B cmd /c "npm start"
                '''
            }
        }

        stage('Wait For App To Boot') {
            steps {
                // Wait ~15 seconds for server to start
                bat 'ping 127.0.0.1 -n 16 > nul'
            }
        }

        stage('Health Check') {
            steps {
                // Fail pipeline if health endpoint not reachable
                bat '''
                curl -f http://localhost:%PORT%/health
                if %errorlevel% neq 0 exit /b 1
                '''
            }
        }
    }

    post {
        success {
            echo 'Pipeline completed successfully!'
        }
        failure {
            echo 'Pipeline failed!'
        }
        always {
            // Kill Node process after pipeline finishes
            bat '''
            taskkill /IM node.exe /F > nul 2>&1
            '''
        }
    }
}
